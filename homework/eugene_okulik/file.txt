number_y = 4

number_c = 4
number_d = 16

# Среднее арифметическое определяется как сумма всех чисел, деленное на их количество
result_1 = (number_c + number_d) / 2

# Среднее геометрическое определяется как умножение всех чисел и извлечение из них корня по количеству чисел
result_2 = (number_c * number_d) / 8

print(result_1)
print(result_2)

number_g = 4
number_h = 16

# Длина гипотенузы прямоугольного треугольника равна сумме квадратов длин катетов
result_1 = (number_g * 4) + (number_h * 16)

# Площадь прямоугольного треугольника равна произведению длин его катетов, разделенному на 2
result_2 = (number_g * number_h) / 2

print(result_1)
print(result_2)
